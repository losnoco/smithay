26/// [`Allocator`] creating [`Dmabuf`]s for system memory via the `udmabuf` uapi.
27///
28/// Note: [`Allocator::create_buffer`] will always return linear buffers for a
29/// [`UdmabufAllocator`] and fail if [`Modifier::Linear`] is not in the list
30/// of passed modifiers.
31#[derive(Debug)]
32pub struct UdmabufAllocator {
33    dev: OwnedFd,
34}
35
36/// Minimum alignment requirement for the stride for `radv` (which seems to be the highest among drivers).
37pub const STRIDE_ALIGN: usize = 256;
38
39impl UdmabufAllocator {
40    /// Tries to open `/dev/udmabuf` and create a [`UdmabufAllocator`].
41    pub fn new() -> io::Result<UdmabufAllocator> {
42        let fd = rustix::fs::open("/dev/udmabuf", OFlags::RDONLY | OFlags::CLOEXEC, Mode::empty())?;
43        Ok(UdmabufAllocator { dev: fd })
44    }
45
46    /// Creates a [`Dmabuf`] from an existing allocation represented by a memfd.
47    ///
48    /// For importing to succeed a few additional requirements need to be satisfied:
49    /// - the width and height cannot exceed 65535.
50    /// - the stride needs to be aligned to [`STRIDE_ALIGN`].
51    /// - size and offset need to be aligned to the platforms page size.
52    ///
53    /// Note: dmabufs created through this interface will always be interpreted as linear.
54    #[allow(clippy::too_many_arguments)]
55    pub fn create_buffer_from_memfd(
56        &self,
57        mem_fd: impl AsFd,
58        offset: usize,
59        size: usize,
60        format: Fourcc,
61        width: u32,
62        height: u32,
63        stride: u32,
64    ) -> Result<Dmabuf, io::Error> {
65        if width >= u16::MAX as u32 || height >= u16::MAX as u32 {
66            return Err(io::ErrorKind::InvalidData.into());
67        }
68        if !(stride as usize).is_multiple_of(STRIDE_ALIGN) {
69            return Err(io::ErrorKind::InvalidData.into());
70        }
71
72        if !size.is_multiple_of(*PAGE_SIZE) || size < (height * stride) as usize {
73            return Err(io::ErrorKind::InvalidData.into());
74        }
75
76        let dma_fd = udmabuf_from_memfd(self.dev.as_fd(), mem_fd.as_fd(), offset as u64, size as u64)?;
77        let mut dmabuf = Dmabuf::builder(
78            Size::new(width as i32, height as i32),
79            format,
80            Modifier::Linear,
81            DmabufFlags::empty(),
82        );
83        dmabuf.add_plane(dma_fd, 0, stride);
84        dmabuf.build().ok_or_else(|| io::ErrorKind::Unsupported.into())
85    }
86}
87
88impl Allocator for UdmabufAllocator {
89    type Buffer = Dmabuf;
90    type Error = io::Error;
91
92    fn create_buffer(
93        &mut self,
94        width: u32,
95        height: u32,
96        fourcc: Fourcc,
97        modifiers: &[Modifier],
98    ) -> Result<Self::Buffer, Self::Error> {
99        if width >= u16::MAX as u32 || height >= u16::MAX as u32 {
100            return Err(io::ErrorKind::InvalidData.into());
101        }
102        if !modifiers.contains(&Modifier::Linear) {
103            return Err(io::ErrorKind::InvalidData.into());
104        }
105        let stride = (width as usize * get_bpp(fourcc).ok_or(io::ErrorKind::InvalidData)?)
106            .next_multiple_of(STRIDE_ALIGN);
107        let size = (height as usize * stride).next_multiple_of(*PAGE_SIZE);
108
109        let mem_fd = memfd_create("udmabuf", MemfdFlags::ALLOW_SEALING | MemfdFlags::CLOEXEC)?;
110        ftruncate(&mem_fd, size as u64)?;
111        fcntl_add_seals(&mem_fd, SealFlags::SHRINK)?;
112        self.create_buffer_from_memfd(mem_fd, 0, size, fourcc, width, height, stride as u32)
113    }
114}
115
116fn udmabuf_from_memfd(
117    fd: BorrowedFd<'_>,
118    mem_fd: BorrowedFd<'_>,
119    offset: u64,
120    size: u64,
121) -> io::Result<OwnedFd> {
122    #[repr(C)]
123    #[derive(Debug)]
124    struct udmabuf_create {
125        memfd: u32,
126        flags: u32,
127        offset: u64,
128        size: u64,
129    }
130    const UDMABUF_FLAGS_CLOEXEC: u32 = 0x01;
131
132    unsafe impl Ioctl for udmabuf_create {
133        type Output = RawFd;
134
135        const IS_MUTATING: bool = false;
136
137        fn opcode(&self) -> Opcode {
138            opcode::write::<udmabuf_create>(b'u', 0x42)
139        }
140
141        fn as_ptr(&mut self) -> *mut rustix::ffi::c_void {
142            self as *mut Self as *mut _
143        }
144
145        unsafe fn output_from_ptr(
146            out: rustix::ioctl::IoctlOutput,
147            _extract_output: *mut rustix::ffi::c_void,
148        ) -> rustix::io::Result<Self::Output> {
149            if out < 0 {
150                Err(rustix::io::Errno::from_raw_os_error(!out))
151            } else {
152                Ok(out)
153            }
154        }
155    }
156
157    let args = udmabuf_create {
158        memfd: mem_fd.as_raw_fd() as u32,
159        flags: UDMABUF_FLAGS_CLOEXEC,
160        offset,
161        size,
162    };
163
164    unsafe {
165        ioctl(fd, args)
166            .map(|fd| OwnedFd::from_raw_fd(fd))
167            .map_err(|err| io::Error::from_raw_os_error(err.raw_os_error()))
168    }
