1//! Module for [dmabuf](https://docs.kernel.org/driver-api/dma-buf.html) buffers.
2//!
3//! `Dmabuf`s act alike to smart pointers and can be freely cloned and passed around.
4//! Once the last `Dmabuf` reference is dropped, its file descriptor is closed and
5//! underlying resources are freed.
6//!
7//! If you want to hold on to a potentially alive dmabuf without blocking the free up
8//! of the underlying resources, you may `downgrade` a `Dmabuf` reference to a `WeakDmabuf`.
9//!
10//! This can be especially useful in resources where other parts of the stack should decide upon
11//! the lifetime of the buffer. E.g. when you are only caching associated resources for a dmabuf.
12

138impl DmabufBuilder {
139    /// Add a plane to the constructed Dmabuf
140    ///
141    /// *Note*: Each Dmabuf needs at least one plane.
142    /// MAX_PLANES notes the maximum amount of planes any format may use with this implementation.
143    pub fn add_plane(&mut self, fd: impl Into<Arc<OwnedFd>>, offset: u32, stride: u32) -> bool {
144        if self.internal.planes.len() == MAX_PLANES {
145            return false;
146        }
147        self.internal.planes.push(Plane {
148            fd: fd.into(),
149            plane_idx: self.internal.planes.len() as u32,
150            offset,
151            stride,
152        });
153
154        true
155    }
156
157    /// Sets a known good node to import the dmabuf with.
158    ///
159    /// While this is only a strong hint and no guarantee, implementations
160    /// should avoid setting this at all, if they can't be reasonably certain.
161    #[cfg(feature = "backend_drm")]
162    pub fn set_node(&mut self, node: DrmNode) {
163        self.internal.node = Mutex::new(Some(node));
164    }
165
166    /// Build a `Dmabuf` out of the provided parameters and planes
167    ///
168    /// Returns `None` if the builder has no planes attached.
169    pub fn build(self) -> Option<Dmabuf> {
170        if self.internal.planes.is_empty() {
171            return None;
172        }
173
174        Some(Dmabuf(Arc::new(self.internal)))
175    }
176}

274    /// Map the plane at specified index with the specified mode
275    ///
276    /// Returns `Err` if the plane with the specified index does not exist or
277    /// mmap failed
278    pub fn map_plane(
279        &self,
280        idx: usize,
281        mode: DmabufMappingMode,
282    ) -> Result<DmabufMapping, DmabufMappingFailed> {
283        let plane = self
284            .0
285            .planes
286            .get(idx)
287            .ok_or(DmabufMappingFailed::PlaneIndexOutOfBound)?;
288
289        let size = rustix::fs::seek(&plane.fd, rustix::fs::SeekFrom::End(0)).map_err(std::io::Error::from)?;
290        rustix::fs::seek(&plane.fd, rustix::fs::SeekFrom::Start(0)).map_err(std::io::Error::from)?;
291
292        let len = (size - plane.offset as u64) as usize;
293        let ptr = unsafe {
294            rustix::mm::mmap(
295                std::ptr::null_mut(),
296                len,
297                mode.into(),
298                rustix::mm::MapFlags::SHARED,
299                &plane.fd,
300                plane.offset as u64,
301            )
302        }
303        .map_err(std::io::Error::from)?;
304        Ok(DmabufMapping { len, ptr })
305    }
306
307    /// Synchronize access for the plane at the specified index
308    ///
309    /// Returns `Err` if the plane with the specified index does not exist or
310    /// the dmabuf_sync ioctl failed
311    pub fn sync_plane(&self, idx: usize, flags: DmabufSyncFlags) -> Result<(), DmabufSyncFailed> {
312        let plane = self
313            .0
314            .planes
315            .get(idx)
316            .ok_or(DmabufSyncFailed::PlaneIndexOutOfBound)?;
317        unsafe { rustix::ioctl::ioctl(&plane.fd, Setter::<DMA_BUF_SYNC, _>::new(dma_buf_sync { flags })) }
318            .map_err(std::io::Error::from)?;
319        Ok(())
320    }
321}

324    /// Modes of mapping a dmabuf plane
325    #[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
326    pub struct DmabufMappingMode: u32 {
327        /// Map the dmabuf readable
328        const READ = 0b00000001;
329        /// Map the dmabuf writable
330        const WRITE = 0b00000010;
331    }
332}
333
334impl From<DmabufMappingMode> for rustix::mm::ProtFlags {
335    #[inline]
336    fn from(mode: DmabufMappingMode) -> Self {
337        let mut flags = rustix::mm::ProtFlags::empty();
338
339        if mode.contains(DmabufMappingMode::READ) {
340            flags |= rustix::mm::ProtFlags::READ;
341        }
342
343        if mode.contains(DmabufMappingMode::WRITE) {
344            flags |= rustix::mm::ProtFlags::WRITE;
345        }
346
347        flags
348    }
349}

499    #[profiling::function]

530#[derive(Debug)]
531enum Subsource {
532    Active(Generic<Arc<OwnedFd>, std::io::Error>),
533    Done(Generic<Arc<OwnedFd>, std::io::Error>),
534    Empty,
535}
536
537impl Subsource {
538    fn done(&mut self) {
539        let mut this = Subsource::Empty;
540        std::mem::swap(self, &mut this);
541        match this {
542            Subsource::Done(source) | Subsource::Active(source) => {
543                *self = Subsource::Done(source);
544            }
545            _ => {}
546        }
547    }

550/// [`Dmabuf`]-based event source. Can be used to monitor implicit fences of a dmabuf.
551#[derive(Debug)]
552pub struct DmabufSource {
553    dmabuf: Dmabuf,
554    signal: Arc<AtomicBool>,
555    sources: [Subsource; 4],
556}
557
558#[derive(thiserror::Error, Debug, Clone, Copy, PartialEq, Eq, Hash)]
559#[error("Dmabuf is already ready for the given interest")]
560/// Dmabuf is already ready for the given interest
561pub struct AlreadyReady;
562
563impl DmabufSource {
564    /// Creates a new [`DmabufSource`] from a [`Dmabuf`] and interest.
565    ///
566    /// This event source will monitor the implicit fences of the given dmabuf.
567    /// Monitoring for READ-access will monitor the state of the most recent write or exclusive fence.
568    /// Monitoring for WRITE-access, will monitor state of all attached fences, shared and exclusive ones.
569    ///
570    /// The event source is a one shot event source and will remove itself from the event loop after being triggered once.
571    /// To monitor for new fences added at a later time a new DmabufSource needs to be created.
572    ///
573    /// Returns `AlreadyReady` if all corresponding fences are already signalled or if `interest` is empty.
574    #[profiling::function]
575    pub fn new(dmabuf: Dmabuf, interest: Interest) -> Result<Self, AlreadyReady> {
576        if !interest.readable && !interest.writable {
577            return Err(AlreadyReady);
578        }
579
580        let mut sources = [
581            Subsource::Empty,
582            Subsource::Empty,
583            Subsource::Empty,
584            Subsource::Empty,
585        ];
586        for (idx, handle) in dmabuf.handles().enumerate() {
587            if matches!(
588                rustix::event::poll(
589                    &mut [rustix::event::PollFd::new(
590                        &handle,
591                        if interest.writable {
592                            rustix::event::PollFlags::OUT
593                        } else {
594                            rustix::event::PollFlags::IN
595                        },
596                    )],
597                    Some(&rustix::time::Timespec {
598                        tv_sec: 0,
599                        tv_nsec: 0
600                    })
601                ),
602                Ok(1)
603            ) {
604                continue;
605            }
606            let fd = dmabuf.0.planes[idx].fd.clone();
607            sources[idx] = Subsource::Active(Generic::new(fd, interest, Mode::OneShot));
608        }
609        if sources
610            .iter()
611            .all(|x| matches!(x, Subsource::Done(_) | Subsource::Empty))
612        {
613            Err(AlreadyReady)
614        } else {
615            Ok(DmabufSource {
616                dmabuf,
617                sources,
618                signal: Arc::new(AtomicBool::new(false)),
619            })
620        }
621    }
622}

624impl EventSource for DmabufSource {
625    type Event = ();
626    type Metadata = Dmabuf;
627    type Ret = Result<(), std::io::Error>;
628
629    type Error = std::io::Error;
630
631    #[profiling::function]
632    fn process_events<F>(
633        &mut self,
634        readiness: calloop::Readiness,
635        token: calloop::Token,
636        mut callback: F,
637    ) -> Result<PostAction, Self::Error>
638    where
639        F: FnMut(Self::Event, &mut Self::Metadata) -> Self::Ret,
640    {
641        for i in 0..4 {
642            if let Ok(PostAction::Remove) = if let Subsource::Active(source) = &mut self.sources[i] {
643                // luckily Generic skips events for other tokens
644                source.process_events(readiness, token, |_, _| Ok(PostAction::Remove))
645            } else {
646                Ok(PostAction::Continue)
647            } {
648                self.sources[i].done();
649            }
650        }
651
652        if self
653            .sources
654            .iter()
655            .all(|x| matches!(x, Subsource::Done(_) | Subsource::Empty))
656        {
657            self.signal.store(true, Ordering::SeqCst);
658            callback((), &mut self.dmabuf)?;
659            Ok(PostAction::Remove)
660        } else {
661            Ok(PostAction::Reregister)
662        }
663    }
664
665    fn register(
666        &mut self,
667        poll: &mut calloop::Poll,
668        token_factory: &mut calloop::TokenFactory,
669    ) -> calloop::Result<()> {
670        for source in self.sources.iter_mut().filter_map(|source| match source {
671            Subsource::Active(source) => Some(source),
672            _ => None,
673        }) {
674            source.register(poll, token_factory)?;
675        }
676        Ok(())
677    }
678
679    fn reregister(
680        &mut self,
681        poll: &mut calloop::Poll,
682        token_factory: &mut calloop::TokenFactory,
683    ) -> calloop::Result<()> {
684        for source in self.sources.iter_mut() {
685            match source {
686                Subsource::Active(source) => source.reregister(poll, token_factory)?,
687                Subsource::Done(source) => {
688                    let _ = source.unregister(poll);
689                }
690                _ => {}
691            }
692        }
693        Ok(())
694    }
695
696    fn unregister(&mut self, poll: &mut calloop::Poll) -> calloop::Result<()> {
697        for source in self.sources.iter_mut() {
698            match source {
699                Subsource::Active(source) => source.unregister(poll)?,
700                Subsource::Done(source) => {
701                    let _ = source.unregister(poll);
702                }
