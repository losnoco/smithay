1//! Types for configuring the color pipeline of a connector.
2//!
3//! These types describe the wide-gamut / HDR signalling state of a DRM connector, i.e. the
4//! values of the optional `Colorspace`, `HDR_OUTPUT_METADATA` and `max bpc` connector
5//! properties. They are used with [`DrmSurface::use_color_state`](super::DrmSurface::use_color_state)
6//! (and the corresponding
7//! [`DrmCompositor::use_color_state`](super::compositor::DrmCompositor::use_color_state)),
8//! which stages the state so that it is applied as part of the *same* atomic commit as the
9//! mode and plane state. Some drivers (notably nvidia) treat a `Colorspace` change as a full
10//! modeset and misbehave when it is committed on its own, so smithay never issues standalone
11//! connector-property commits for these.
12//!
13//! Whether a connector supports these properties can be queried with
14//! [`DrmSurface::supported_colorspaces`](super::DrmSurface::supported_colorspaces),
15//! [`DrmSurface::hdr_metadata_supported`](super::DrmSurface::hdr_metadata_supported) and
16//! [`DrmSurface::max_bpc_range`](super::DrmSurface::max_bpc_range).
17//!
18//! Capabilities of the connected sink (whether it accepts a PQ EOTF, its desired luminance
19//! range, BT.2020 signal support) should be read from its EDID, e.g. via
20//! `smithay-drm-extras`' `display_info` module and libdisplay-info's
21//! `Info::hdr_static_metadata()` / `Info::supported_signal_colorimetry()`.
22
23/// Value of the `Colorspace` connector property.
24///
25/// This selects the colorimetry signalled to the sink in the AVI infoframe (HDMI) or MSA/SDP
26/// (DisplayPort). It does not perform any color conversion; the submitted framebuffer contents
27/// are expected to already be encoded in the signalled colorspace.
28///
29/// Only a portable subset of the kernel's colorspace values is exposed. The property's enum
30/// values are driver-defined and are resolved by name at set time.
31#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
32pub enum Colorspace {
33    /// The default colorspace of the connector (`Default`), typically sRGB/BT.709.
34    #[default]
35    Default,
36    /// ITU-R BT.2020 RGB colorimetry (`BT2020_RGB`).
37    Bt2020Rgb,
38    /// ITU-R BT.2020 YCbCr colorimetry (`BT2020_YCC`).
39    Bt2020Ycc,
40    /// DCI-P3 RGB colorimetry with D65 white point (`DCI-P3_RGB_D65`).
41    DciP3RgbD65,
42    /// A colorspace not modelled by smithay.
43    ///
44    /// Only ever returned when reading back the current state of a connector; requesting it
45    /// in [`use_color_state`](super::DrmSurface::use_color_state) fails.
46    Unknown,
47}
48
49impl Colorspace {
50    /// The kernel's name for this colorspace in the `Colorspace` property enum.
51    ///
52    /// Returns `None` for [`Colorspace::Unknown`].
53    pub fn kernel_name(&self) -> Option<&'static str> {
54        Some(match self {
55            Colorspace::Default => "Default",
56            Colorspace::Bt2020Rgb => "BT2020_RGB",
57            Colorspace::Bt2020Ycc => "BT2020_YCC",
58            Colorspace::DciP3RgbD65 => "DCI-P3_RGB_D65",
59            Colorspace::Unknown => return None,
60        })
61    }
62
63    pub(super) fn from_kernel_name(name: &str) -> Option<Self> {
64        Some(match name {
65            "Default" => Colorspace::Default,
66            "BT2020_RGB" => Colorspace::Bt2020Rgb,
67            "BT2020_YCC" => Colorspace::Bt2020Ycc,
68            "DCI-P3_RGB_D65" => Colorspace::DciP3RgbD65,
69            _ => return None,
70        })
71    }
72}

74/// Electro-optical transfer function of the HDR metadata infoframe, per CTA-861.3.
75#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
76pub enum Eotf {
77    /// Traditional gamma, SDR luminance range.
78    TraditionalSdr,
79    /// Traditional gamma, HDR luminance range.
80    TraditionalHdr,
81    /// SMPTE ST 2084, a.k.a. Perceptual Quantizer (PQ).
82    SmpteSt2084,
83    /// Hybrid Log-Gamma (HLG), per BT.2100.
84    Hlg,
85}
86
87impl Eotf {
88    fn to_raw(self) -> u8 {
89        match self {
90            Eotf::TraditionalSdr => 0,
91            Eotf::TraditionalHdr => 1,
92            Eotf::SmpteSt2084 => 2,
93            Eotf::Hlg => 3,
94        }
95    }
96
97    fn from_raw(raw: u8) -> Option<Self> {
98        Some(match raw {
99            0 => Eotf::TraditionalSdr,
100            1 => Eotf::TraditionalHdr,
101            2 => Eotf::SmpteSt2084,
102            3 => Eotf::Hlg,
103            _ => return None,
104        })
105    }
106}

108/// A CIE 1931 xy chromaticity coordinate in CTA-861.3 units, i.e. the floating-point
109/// coordinate scaled by 50000 (increments of 0.00002).
110#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
111pub struct CtaCoordinate {
112    /// The x coordinate, scaled by 50000.
113    pub x: u16,
114    /// The y coordinate, scaled by 50000.
115    pub y: u16,
116}
117
118impl CtaCoordinate {
119    /// Converts a floating point CIE 1931 xy coordinate into CTA-861.3 units.
120    pub fn from_xy(x: f64, y: f64) -> Self {
121        Self {
122            x: (x * 50000.0).round() as u16,
123            y: (y * 50000.0).round() as u16,
124        }
125    }
126
127    /// The BT.2020 red primary (0.708, 0.292).
128    pub const BT2020_RED: Self = Self { x: 35400, y: 14600 };
129    /// The BT.2020 green primary (0.170, 0.797).
130    pub const BT2020_GREEN: Self = Self { x: 8500, y: 39850 };
131    /// The BT.2020 blue primary (0.131, 0.046).
132    pub const BT2020_BLUE: Self = Self { x: 6550, y: 2300 };
133    /// The D65 white point (0.3127, 0.3290).
134    pub const D65_WHITE: Self = Self { x: 15635, y: 16450 };
135}

137/// Static HDR metadata (CTA-861.3 Static Metadata Type 1) for the `HDR_OUTPUT_METADATA`
138/// connector property.
139///
140/// This describes the mastering display and content light levels of the submitted frames to
141/// the sink. All fields use the raw infoframe units, so no rounding is hidden from callers.
142#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
143pub struct HdrOutputMetadata {
144    /// The electro-optical transfer function the frame contents are encoded with.
145    pub eotf: Eotf,
146    /// Chromaticity of the mastering display's red, green and blue primaries.
147    pub display_primaries: [CtaCoordinate; 3],
148    /// Chromaticity of the mastering display's white point.
149    pub white_point: CtaCoordinate,
150    /// Maximum luminance of the mastering display, in cd/m².
151    pub max_display_mastering_luminance: u16,
152    /// Minimum luminance of the mastering display, in 0.0001 cd/m² units.
153    pub min_display_mastering_luminance: u16,
154    /// Maximum content light level, in cd/m².
155    pub max_cll: u16,
156    /// Maximum frame-average light level, in cd/m².
157    pub max_fall: u16,
158}
159
160impl HdrOutputMetadata {
161    /// Convenience constructor for the most common HDR10-style signal: PQ transfer function
162    /// with BT.2020 mastering primaries and a D65 white point.
163    ///
164    /// Luminance values should be clamped to what the sink advertises in its EDID HDR static
165    /// metadata block.
166    pub fn pq_bt2020(max_luminance: u16, min_luminance: u16, max_cll: u16, max_fall: u16) -> Self {
167        Self {
168            eotf: Eotf::SmpteSt2084,
169            display_primaries: [
170                CtaCoordinate::BT2020_RED,
171                CtaCoordinate::BT2020_GREEN,
172                CtaCoordinate::BT2020_BLUE,
173            ],
174            white_point: CtaCoordinate::D65_WHITE,
175            max_display_mastering_luminance: max_luminance,
176            min_display_mastering_luminance: min_luminance,
177            max_cll,
178            max_fall,
179        }
180    }
181}

183/// Desired color pipeline configuration of a connector.
184///
185/// The default value describes plain SDR signalling: default colorspace, no HDR metadata and
186/// the `max bpc` property left untouched.
187#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
188pub struct ConnectorColorState {
189    /// The colorimetry to signal via the `Colorspace` property.
190    pub colorspace: Colorspace,
191    /// The HDR static metadata to signal via the `HDR_OUTPUT_METADATA` property.
192    ///
193    /// `None` disables the HDR infoframe (the property is set to no blob).
194    pub hdr_metadata: Option<HdrOutputMetadata>,
195    /// The maximum bits per component to allow on the link via the `max bpc` property.
196    ///
197    /// `None` leaves the property at its current value.
198    pub max_bpc: Option<u32>,
199}

201pub(super) mod ffi {
202    //! Binary layout of the kernel's `HDR_OUTPUT_METADATA` blob
203    //! (`struct hdr_output_metadata` in `include/uapi/drm/drm_mode.h`).
204
205    /// Static Metadata Type 1 descriptor id (CTA-861.3).
206    const HDMI_STATIC_METADATA_TYPE1: u32 = 0;
207
208    #[repr(C)]
209    #[derive(Debug, Clone, Copy)]
210    pub struct HdrColorPoint {
211        pub x: u16,
212        pub y: u16,
213    }
214
215    /// `struct hdr_metadata_infoframe`.
216    #[repr(C)]
217    #[derive(Debug, Clone, Copy)]
218    pub struct HdrMetadataInfoframe {
219        pub eotf: u8,
220        pub metadata_type: u8,
221        pub display_primaries: [HdrColorPoint; 3],
222        pub white_point: HdrColorPoint,
223        pub max_display_mastering_luminance: u16,
224        pub min_display_mastering_luminance: u16,
225        pub max_cll: u16,
226        pub max_fall: u16,
227    }
228
229    /// `struct hdr_output_metadata`. Passed to the kernel verbatim as the blob contents,
230    /// so the layout must match exactly.
231    #[repr(C)]
232    #[derive(Debug, Clone, Copy)]
233    pub struct HdrOutputMetadata {
234        pub metadata_type: u32,
235        pub hdmi_metadata_type1: HdrMetadataInfoframe,
236    }
237
238    impl From<super::HdrOutputMetadata> for HdrOutputMetadata {
239        fn from(meta: super::HdrOutputMetadata) -> Self {
240            let coord = |c: super::CtaCoordinate| HdrColorPoint { x: c.x, y: c.y };
241            HdrOutputMetadata {
242                metadata_type: HDMI_STATIC_METADATA_TYPE1,
243                hdmi_metadata_type1: HdrMetadataInfoframe {
244                    eotf: meta.eotf.to_raw(),
245                    metadata_type: HDMI_STATIC_METADATA_TYPE1 as u8,
246                    display_primaries: [
247                        coord(meta.display_primaries[0]),
248                        coord(meta.display_primaries[1]),
249                        coord(meta.display_primaries[2]),
250                    ],
251                    white_point: coord(meta.white_point),
252                    max_display_mastering_luminance: meta.max_display_mastering_luminance,
253                    min_display_mastering_luminance: meta.min_display_mastering_luminance,
254                    max_cll: meta.max_cll,
255                    max_fall: meta.max_fall,
256                },
257            }
258        }
259    }
260
261    impl HdrOutputMetadata {
262        /// Parses the contents of an `HDR_OUTPUT_METADATA` blob.
263        ///
264        /// Returns `None` if the blob is too short, describes a different metadata type or
265        /// uses an EOTF unknown to us.
266        pub fn parse(bytes: &[u8]) -> Option<super::HdrOutputMetadata> {
267            // The infoframe is valid without the trailing struct padding, so only require the
268            // payload itself.
269            const PAYLOAD_LEN: usize = 30;
270            if bytes.len() < PAYLOAD_LEN {
271                return None;
272            }
273            let u16_at = |off: usize| u16::from_le_bytes([bytes[off], bytes[off + 1]]);
274            let coord_at = |off: usize| super::CtaCoordinate {
275                x: u16_at(off),
276                y: u16_at(off + 2),
277            };
278
279            let metadata_type = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
280            if metadata_type != HDMI_STATIC_METADATA_TYPE1 {
281                return None;
282            }
283
284            Some(super::HdrOutputMetadata {
285                eotf: super::Eotf::from_raw(bytes[4])?,
286                display_primaries: [coord_at(6), coord_at(10), coord_at(14)],
287                white_point: coord_at(18),
288                max_display_mastering_luminance: u16_at(22),
289                min_display_mastering_luminance: u16_at(24),
290                max_cll: u16_at(26),
291                max_fall: u16_at(28),
292            })
293        }
294    }
295}

301    #[test]
302    fn hdr_output_metadata_layout() {
303        // The blob is passed to the kernel verbatim, so the layout must match
304        // `struct hdr_output_metadata` exactly: 4-byte type + 26-byte infoframe, padded to 32.
305        assert_eq!(std::mem::size_of::<ffi::HdrOutputMetadata>(), 32);
306
307        let meta: ffi::HdrOutputMetadata = HdrOutputMetadata::pq_bt2020(500, 50, 400, 300).into();
308        let bytes = unsafe {
309            std::slice::from_raw_parts(
310                &meta as *const ffi::HdrOutputMetadata as *const u8,
311                std::mem::size_of::<ffi::HdrOutputMetadata>(),
312            )
313        };
314
315        // metadata_type: u32 = HDMI_STATIC_METADATA_TYPE1 (0)
316        assert_eq!(&bytes[0..4], &0u32.to_le_bytes());
317        // infoframe.eotf = SMPTE ST 2084 (PQ) = 2
318        assert_eq!(bytes[4], 2);
319        // infoframe.metadata_type = 0 (Static Metadata Type 1)
320        assert_eq!(bytes[5], 0);
321        // display_primaries[0] (red) = BT.2020 (0.708, 0.292) * 50000
322        assert_eq!(&bytes[6..8], &35400u16.to_le_bytes());
323        assert_eq!(&bytes[8..10], &14600u16.to_le_bytes());
324        // display_primaries[1] (green) = (0.170, 0.797) * 50000
325        assert_eq!(&bytes[10..12], &8500u16.to_le_bytes());
326        assert_eq!(&bytes[12..14], &39850u16.to_le_bytes());
327        // display_primaries[2] (blue) = (0.131, 0.046) * 50000
328        assert_eq!(&bytes[14..16], &6550u16.to_le_bytes());
329        assert_eq!(&bytes[16..18], &2300u16.to_le_bytes());
330        // white_point = D65 (0.3127, 0.3290) * 50000
331        assert_eq!(&bytes[18..20], &15635u16.to_le_bytes());
332        assert_eq!(&bytes[20..22], &16450u16.to_le_bytes());
333        // luminances and light levels
334        assert_eq!(&bytes[22..24], &500u16.to_le_bytes());
335        assert_eq!(&bytes[24..26], &50u16.to_le_bytes());
336        assert_eq!(&bytes[26..28], &400u16.to_le_bytes());
337        assert_eq!(&bytes[28..30], &300u16.to_le_bytes());
338    }

343        let raw: ffi::HdrOutputMetadata = meta.into();
344        let bytes = unsafe {
